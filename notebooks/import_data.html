# define a function for sweeps extraction
extract_sweep <- function(sweep_n, abf) {
  abf %>%
    as.data.frame(sweep = sweep_n) %>% 
    mutate(sweep = sweep_n)
}

# define a function to extract all sweeps from different files
abf_to_df <- function(abf) {
  map(1:length(abf$data), extract_sweep, abf = abf) %>%
    bind_rows()

3 Now import all the data from the cell

# list the files from the target directory
files <- list.files(
  "../data/2023-03-07", 
  pattern = ".abf",  
  full.names = TRUE
  )

# read all files in the directory
abfs <- map(files, readABF)
# convert files into data frames
dfs <- map(abfs, abf_to_df)

# extract file subname to change naming of dataframes
files |> str_sub(-9, -5)
[1] "_0001" "_0002" "_0004" "_0005" "_0007" "_0008" "_0011"
# name dataframes corresponding to their file
names(dfs) <- str_sub(files, -9, -5)

# make better column names (on example)
invisible(example_df |> rename(
  time_s = `Time [s]`,
  current_pA = any_of(c("Currnt0Pr [pA]", "Volt0Pr [pA]"))
))
datatable(head(example_df, n=15))
# make better column names on all dataframes
dfs <- map(
    dfs, 
    rename,  
    time_s = `Time [s]`,  
    current_pA = any_of(c("Currnt0Pr [pA]", "Volt0Pr [pA]"))
    )
# select only necessary columns (drop voltage)
dfs <- map(dfs, select, time_s, current_pA, sweep)

# merge all dataframes into single one 
# and introduce file name variable
all_sweeps <- bind_rows(dfs, .id = "file")

4 Add parsed annotation from .txt file

Here we add the variables for stimulation tyoe and condition (control, noradrenaline, washout)

all_sweeps <- all_sweeps %>%
    # only the files with AC or C stim, there is no recordings of 
    # A-stim in Noradrenaline so no need to import it
    filter (file %in% c("_0001", "_0002", "_0007", "_0011")) %>%
    mutate(
        # type of fibers stimulated in this file / sweep (C or both A and C)
        stim = case_when(
            file %in% c("_0001", "_0002") ~ if_else(sweep <= 10, "AC", "C"),
            file %in% c("_0007", "_0011") ~ if_else(sweep <= 7, "AC", "C")
        ),
        # condition of recording (control, noradrenaline, washout)
        cond = case_when(
            file %in% c("_0001", "_0002", "_0007") ~ "control",
            file %in% c("_0011") ~ "noradrenalin"
        )
    ) %>%
    mutate(
        stim = as.factor(stim),
        cond = as.factor(cond),
    )
